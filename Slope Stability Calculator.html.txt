    <h1>Slope Stability Calculator</h1>
    <form>
        <label>Weight of Slope (W) [kN]: <input type="number" id="W" required></label><br>
        <label>Frictional Resistance (R) [kN]: <input type="number" id="R" required></label><br>
        <button type="button" onclick="calculateSlopeStability()">Calculate</button>
    </form>
    <p id="result"></p>

    <script>
        function calculateSlopeStability() {
            const W = parseFloat(document.getElementById('W').value);
            const R = parseFloat(document.getElementById('R').value);

            const FS = R / W; // Factor of Safety
            document.getElementById('result').innerText = `Factor of Safety: ${FS.toFixed(2)}`;
        }
    </script>
